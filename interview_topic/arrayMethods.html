JavaScript Array Methods Reference

1. push()
Adds elements to end
let arr = [1, 2]; arr.push(3, 4); // Result: [1, 2, 3, 4]
2. pop()
Removes last element
let arr = [1, 2, 3]; let last = arr.pop(); // last = 3, arr = [1, 2]
3. unshift()
Adds elements to beginning
let arr = [2, 3]; arr.unshift(0, 1); // Result: [0, 1, 2, 3]
4. shift()
Removes first element
let arr = [1, 2, 3]; let first = arr.shift(); // first = 1, arr = [2, 3]
5. map()
Transform each element
let nums = [1, 2, 3]; let doubled = nums.map(n => n * 2); // Result: [2, 4, 6]
6. filter()
Filter elements by condition
let nums = [1, 2, 3, 4]; let even = nums.filter(n => n % 2 === 0); // Result: [2, 4]
7. reduce()
Reduce to single value
let nums = [1, 2, 3, 4]; let sum = nums.reduce((a, b) => a + b, 0); // Result: 10
8. find()
Find first matching element
let users = [{id: 1}, {id: 2}]; let found = users.find(u => u.id === 2); // Result: {id: 2}
9. findIndex()
Find index of first match
let arr = [10, 20, 30]; let index = arr.findIndex(n => n > 15); // Result: 1
10. includes()
Check if element exists
let fruits = ['apple', 'banana']; fruits.includes('banana'); // Result: true
11. some()
Check if any element passes test
let nums = [1, 3, 5]; nums.some(n => n % 2 === 0); // Result: false
12. every()
Check if all elements pass test
let nums = [2, 4, 6]; nums.every(n => n % 2 === 0); // Result: true
13. sort()
Sort array in place
let nums = [4, 1, 10, 2]; nums.sort((a, b) => a - b); // Result: [1, 2, 4, 10]
14. reverse()
Reverse array in place
let arr = [1, 2, 3]; arr.reverse(); // Result: [3, 2, 1]
15. splice()
Add/remove elements
let arr = [1, 2, 3, 4]; arr.splice(1, 2); // Result: [1, 4]
16. slice()
Copy portion of array
let arr = [10, 20, 30, 40]; let part = arr.slice(1, 3); // Result: [20, 30]
17. concat()
Combine arrays
let a = [1, 2]; let b = [3, 4]; let result = a.concat(b); // Result: [1, 2, 3, 4]
18. flat()
Flatten nested arrays
let arr = [1, [2, [3]]]; arr.flat(2); // Result: [1, 2, 3]
19. join()
Join elements to string
let arr = ['a', 'b', 'c']; arr.join('-'); // Result: 'a-b-c'
20. at()
Access with negative index
let arr = [10, 20, 30]; arr.at(-1); // Result: 30
